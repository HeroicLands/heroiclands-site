Attributes auto-fit grid of label+value pairs 6. Skills one paragraph per category, small-caps inline list Display-name and category mappings live in data/sohl.yaml. Every section is guarded so missing data silently disappears. */ -}} {{- $sohl := site.Data.sohl -}} {{- $attrNames := $sohl.attributes -}} {{- $attrCols := $sohl.attribute_columns -}} {{- $skillNames := $sohl.skills -}} {{- $categories := $sohl.skill_categories -}} {{- $attrs := .Params.sohl.attributes -}} {{- $charSkills := .Params.sohl.skills -}} {{- $traits := .Params.traits -}} {{- $appearance := "" -}} {{- $birthsign := "" -}} {{- with .Params.sohl }}{{ $birthsign = .birthsign }}{{ end -}}
{{- /* ---- Portrait ---- */ -}} {{- with .Params.slug }} {{- end }}

{{- /* ---- Appearance (composed sentence) ---- */ -}} {{- $parts := slice -}} {{- with $traits.age }}{{ $parts = $parts | append (printf "Age %v" .) }}{{ end -}} {{- with $traits.height }}{{- with .m -}} {{- $totalInches := math.Round (mul . 39.3701) -}} {{- $feet := int (math.Floor (div $totalInches 12)) -}} {{- $inches := int (sub $totalInches (mul $feet 12)) -}} {{- if eq $inches 12 }}{{ $feet = add $feet 1 }}{{ $inches = 0 }}{{ end -}} {{- $parts = $parts | append (printf "%d′ %d″" $feet $inches) -}} {{- end }}{{- end -}} {{- with $traits.weight }}{{- with .kg -}} {{- $lbs := int (math.Round (mul . 2.20462)) -}} {{- $parts = $parts | append (printf "%d lbs" $lbs) -}} {{- end }}{{- end -}} {{- with $traits.build }}{{- with .frame }}{{ $parts = $parts | append (printf "%s frame" .) }}{{ end }}{{ end -}} {{- with $traits.appearance -}} {{- with .eye_color }}{{ $parts = $parts | append (printf "%s eyes" .) }}{{ end -}} {{- with .hair_color }}{{ $parts = $parts | append (printf "%s hair" .) }}{{ end -}} {{- with .skin_color }}{{ $parts = $parts | append (printf "%s skin" .) }}{{ end -}} {{- end -}} {{- if gt (len $parts) 0 }}

Appearance: {{ delimit $parts ", " | safeHTML }}

{{- end }} {{- /* ---- Attributes ---- */ -}} {{- $attrOrder := slice -}} {{- range $attrCols.left }}{{ $attrOrder = $attrOrder | append . }}{{ end -}} {{- range $attrCols.right }}{{ $attrOrder = $attrOrder | append . }}{{ end -}} {{- $hasAnyAttr := false -}} {{- with $attrs }}{{ $hasAnyAttr = gt (len .) 0 }}{{ end -}} {{- if $hasAnyAttr }}

{{- range $code := $attrOrder }} {{- $score := "" }}{{ with $attrs }}{{ $score = index . $code }}{{ end }} {{- $label := index $attrNames $code | default (upper $code) }}

{{- end }}
{{- end }} {{- /* ---- Skills (grouped by category) ---- */ -}} {{- $hasAnySkill := false -}} {{- with $charSkills }}{{ $hasAnySkill = gt (len .) 0 }}{{ end -}} {{- if $hasAnySkill }}

Skills

{{- /* Build a set of categorized codes so we can detect uncategorized leftovers. */ -}} {{- $categorized := dict -}} {{- range $categories -}} {{- range .skills -}}{{ $categorized = merge $categorized (dict . true) }}{{- end -}} {{- end -}} {{- range $categories }} {{- $catLabel := .label -}} {{- $entries := slice -}} {{- range $code := .skills -}} {{- $score := "" -}} {{- with $charSkills }}{{ $score = index . $code }}{{ end -}} {{- with $score -}} {{- $name := index $skillNames $code | default $code -}} {{- $entries = $entries | append (dict "name" $name "score" .) -}} {{- end -}} {{- end -}} {{- if gt (len $entries) 0 }}

{{ $catLabel }}: {{- range $i, $e := $entries -}} {{- if $i -}}, {{ end -}} {{ $e.name }} {{ $e.score }} {{- end -}}

{{- end }} {{- end }} {{- /* Catch any skills the character has that aren't in any category. */ -}} {{- $other := slice -}} {{- range $code, $score := $charSkills -}} {{- if not (index $categorized $code) -}} {{- $name := index $skillNames $code | default $code -}} {{- $other = $other | append (dict "name" $name "score" $score) -}} {{- end -}} {{- end -}} {{- if gt (len $other) 0 -}} {{- $other = sort $other "name" }}

Other: {{- range $i, $e := $other -}} {{- if $i -}}, {{ end -}} {{ $e.name }} {{ $e.score }} {{- end -}}

{{- end }}
