{{- /* Creature info sidebar — profile-card style. Layout: 1. Creature image (https://cdn.heroiclands.org/images/{slug}.webp) 2. "Creature Profile" header 3. Profile header dl (name / lineage) 4. Attributes auto-fit grid of label+value pairs (only attributes the creature actually has are rendered; creature stat blocks are typically sparser than character ones) 5. Skills one paragraph per category, small-caps inline list Display-name and category mappings live in data/sohl.yaml and are shared with the character sidebar. Every section is guarded so missing data silently disappears. */ -}} {{- $sohl := hugo.Data.sohl -}} {{- $attrNames := $sohl.attributes -}} {{- $attrCols := $sohl.attribute_columns -}} {{- $skillNames := $sohl.skills -}} {{- $categories := $sohl.skill_categories -}} {{- $attrs := dict -}}{{- with .Params.sohl }}{{- with .attributes }}{{ $attrs = . }}{{ end }}{{ end -}} {{- $cSkills := dict -}}{{- with .Params.sohl }}{{- with .skills }}{{ $cSkills = . }}{{ end }}{{ end -}} {{- $lineage := "" -}}{{- with .Params.traits }}{{- with .lineage }}{{ $lineage = . }}{{ end }}{{ end -}}

{{- range $code := $attrOrder }} {{- $score := "" -}} {{- with $attrs }}{{ $score = index . $code }}{{ end -}} {{- with $score }} {{- $label := index $attrNames $code | default (upper $code) }}

Skills

{{- /* Build a set of categorized codes so we can detect uncategorized leftovers. */ -}} {{- $categorized := dict -}} {{- range $categories -}} {{- range .skills -}}{{ $categorized = merge $categorized (dict . true) }}{{- end -}} {{- end -}} {{- range $categories }} {{- $catLabel := .label -}} {{- $entries := slice -}} {{- range $code := .skills -}} {{- $score := "" -}} {{- with $cSkills }}{{ $score = index . $code }}{{ end -}} {{- with $score -}} {{- $name := index $skillNames $code | default $code -}} {{- $entries = $entries | append (dict "name" $name "score" .) -}} {{- end -}} {{- end -}} {{- if gt (len $entries) 0 }}

{{ $catLabel }}: {{- range $i, $e := $entries -}} {{- if $i -}}, {{ end -}} {{ $e.name }} {{ $e.score }} {{- end -}}

{{- end }} {{- end }} {{- /* Catch any skills the creature has that aren't in any category. */ -}} {{- $other := slice -}} {{- range $code, $score := $cSkills -}} {{- if not (index $categorized $code) -}} {{- $name := index $skillNames $code | default $code -}} {{- $other = $other | append (dict "name" $name "score" $score) -}} {{- end -}} {{- end -}} {{- if gt (len $other) 0 -}} {{- $other = sort $other "name" }}

Other: {{- range $i, $e := $other -}} {{- if $i -}}, {{ end -}} {{ $e.name }} {{ $e.score }} {{- end -}}

{{- end }}
