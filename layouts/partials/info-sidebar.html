{{- /* Info sidebar dispatcher. Renders a type-specific structured-info sidebar alongside the page's main content. Types without a sidebar partial render nothing; the single.html layout detects this and skips reserving column space. To add a new type: create layouts/partials/sidebars/{type}.html and add a branch below. */ -}} {{- $type := .Params.type | default "" -}} {{- $gearTypes := slice "weapongear" "armorgear" "miscgear" "containergear" "projectilegear" "concoctiongear" -}} {{- if eq $type "character" -}} {{- partial "sidebars/character.html" . -}} {{- else if eq $type "creature" -}} {{- partial "sidebars/creature.html" . -}} {{- else if in $gearTypes $type -}} {{- partial "sidebars/gear.html" . -}} {{- end -}}
