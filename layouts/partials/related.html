{{- /* Related partial. Renders a compact "Related" block at the bottom of a single page using the backlinks and mentions lists injected into frontmatter by the Hugo exporter (scripts/export-hugo.ts). Structure in frontmatter: related: backlinks: [{ title, url, type }, ...] # pages that link here mentions: [{ title, url, type }, ...] # pages this page links to Rendering strategy: - Merge both directions into a single deduped pool (direction doesn't matter to a reader — a connection is a connection). - Group by the referenced page's `type`. - Emit one line per type: "Label: link • link • link". - The whole block is wrapped in a bordered, tinted card so it visually sets off the end of the article. */ -}} {{- $rel := .Params.related -}} {{- if not $rel -}}{{- return -}}{{- end -}} {{- $backlinks := or $rel.backlinks slice -}} {{- $mentions := or $rel.mentions slice -}} {{- if and (eq (len $backlinks) 0) (eq (len $mentions) 0) -}} {{- return -}} {{- end -}} {{- /* Merge both directions, deduped by URL. `union` can't hash slices of maps, so we walk the two lists in order and dedupe by URL manually. Mentions come first so explicit author choices win when a page appears in both directions. */ -}} {{- $seen := slice -}} {{- $merged := slice -}} {{- range $mentions -}} {{- if not (in $seen .url) -}} {{- $seen = $seen | append .url -}} {{- $merged = $merged | append . -}} {{- end -}} {{- end -}} {{- range $backlinks -}} {{- if not (in $seen .url) -}} {{- $seen = $seen | append .url -}} {{- $merged = $merged | append . -}} {{- end -}} {{- end -}} {{- /* Plural, human-readable label for each type. Falls back to `humanize` for any type not listed here. */ -}} {{- $labels := dict "blog-post" "Posts" "character" "People" "continent" "Continents" "creature" "Creatures" "faith" "Faiths" "language" "Languages" "lore" "Lore" "organization" "Organizations" "page" "Pages" "pantheon" "Pantheons" "people" "Peoples" "polity" "Polities" "project-page" "Projects" "region" "Regions" "religion" "Religions" "settlement" "Settlements" "site" "Places" "world" "Worlds" -}} {{- /* Group refs by type, preserving alphabetical order within a group (input is already sorted by the exporter). */ -}} {{- $groups := dict -}} {{- range $merged -}} {{- $t := .type | default "page" -}} {{- $existing := index $groups $t | default slice -}} {{- $groups = merge $groups (dict $t ($existing | append .)) -}} {{- end -}} {{- /* Sort the group keys alphabetically by their rendered label so the list reads A → Z regardless of insertion order. */ -}} {{- $rows := slice -}} {{- range $type, $refs := $groups -}} {{- $label := index $labels $type | default (humanize $type) -}} {{- $rows = $rows | append (dict "label" $label "type" $type "refs" $refs) -}} {{- end -}} {{- $rows = sort $rows "label" "asc" -}}

Related

{{- range $rows }}
{{ .label }}
{{- range $i, $ref := .refs -}} {{- if $i -}} • {{- end -}} {{ $ref.title }} {{- end -}}
{{- end }}
