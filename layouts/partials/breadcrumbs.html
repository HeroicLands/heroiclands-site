{{- /* Breadcrumbs partial. Renders `Home > {section label} > {name}` using the page's type and package frontmatter. Behavior: - Home page: not rendered. - Collection section landings (type == "doc", category == "collection", URL == /{pkg}/{slug}/) and Hugo's auto-generated section indexes: Home > {Title} since the page itself *is* the section. - Package-driven leaf content (type == "doc" OR any type that is package-routed by the exporter): Middle crumb composed from .Params.package and either .Params.category (for doc) or the type itself (for everything else). URL = /{package}/{middle}/. Label = $packageLabels[package] + " " + titleCase(middle). Keep $packageLabels in sync with the packages actually used in the vault. - Compound types under /hm3/ ({system}-{kind}, e.g. hm3-user-guide) get a curated label and link from $compoundLabels. - Pages with no type declared (e.g., blog/project landings, auto- generated section indexes): Home > {Title} */ -}} {{- if .IsHome -}}{{- return -}}{{- end -}} {{- $type := .Params.type | default "" -}} {{- $category := .Params.category | default "" -}} {{- $isCollection := and (eq $type "doc") (eq $category "collection") -}} {{- /* Curated labels for compound types under /hm3/ */ -}} {{- $compoundLabels := dict "hm3-rules" (dict "label" "HM3 Rules" "url" "/hm3/rules/") "hm3-user-guide" (dict "label" "HM3 User Guide" "url" "/hm3/user-guide/") -}} {{- /* Proper-noun labels for packages. Title-case alone can't produce "SoHL" or "HM3", so each known package gets an explicit label here. Unknown packages fall back to title-casing the package slug. */ -}} {{- $packageLabels := dict "sohl" "SoHL" "thalorna" "Thalorna" "kethira" "Kethira" "hm3" "HM3" -}}
  1. Home
    2. {{- if and (ne $type "") (not $isCollection) -}} {{- $section := "" -}} {{- $compound := index $compoundLabels $type -}} {{- if $compound -}} {{- $section = $compound -}} {{- else -}} {{- /* Package-driven: middle = category for doc, otherwise the type. */ -}} {{- $pkg := .Params.package | default "" -}} {{- $middle := "" -}} {{- if eq $type "doc" -}} {{- $middle = $category -}} {{- else -}} {{- $middle = $type -}} {{- end -}} {{- if and (ne $pkg "") (ne $middle "") -}} {{- $pkgLabel := index $packageLabels $pkg | default (title $pkg) -}} {{- $middleLabel := replace $middle "-" " " | title -}} {{- $section = dict "label" (printf "%s %s" $pkgLabel $middleLabel) "url" (printf "/%s/%s/" $pkg $middle) -}} {{- end -}} {{- end -}} {{- if $section -}}
  2. {{ $section.label }}
    3. {{- else -}}
  3. {{ $type }}
    4. {{- end -}} {{- end -}}
  4. {{ .Title }}
