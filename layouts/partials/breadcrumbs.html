{{- /* Breadcrumbs partial. Renders `Home > {section label} > {name}` using the page's type field, and — for type=doc — its package and category properties, plus the canonical Types catalog in /type/. Behavior: - Home page: not rendered. - Aggregation pages (type == "type-catalog" or "section"): Home > {Title} with no middle segment, since the page *is* an aggregator. - Documentation pages (type == "doc"): The middle crumb is composed from .Params.package and .Params.category. URL = /{package}/{category}/. Label = $packageLabels[package] + " " + titleCase(category). The same properties drive URL routing in scripts/export-hugo.ts; keep $packageLabels in sync with the packages used in the vault. - Compound types under /hm3/ ({system}-{kind}, e.g. hm3-user-guide) get a curated label and link to the appropriate section URL — no /type/{type}/ page needed. - Typed content pages where /type/{type}/ exists: Home > {linked label} > {Title} with the label pulled from the catalog page's Title. - Typed content pages with no matching /type/{type}/ page: Home > {type slug as plain text} > {Title} - Pages with no type declared: Home > {Title} The /type/{type}/ lookup uses the slug set in each Types/*.md file's frontmatter, not the filename, so the singular-type / plural-title split is transparent here. */ -}} {{- if .IsHome -}}{{- return -}}{{- end -}} {{- $type := .Params.type | default "" -}} {{- $isAggregator := or (eq $type "type-catalog") (eq $type "section") -}} {{- /* Curated labels for compound types that don't have /type/{type}/ pages */ -}} {{- $compoundLabels := dict "hm3-rules" (dict "label" "HM3 Rules" "url" "/hm3/rules/") "hm3-user-guide" (dict "label" "HM3 User Guide" "url" "/hm3/user-guide/") -}} {{- /* Proper-noun labels for doc packages. Title-case alone can't produce "SoHL" or "HM3", so each known package gets an explicit label here. Unknown packages fall back to title-casing the package slug. */ -}} {{- $packageLabels := dict "sohl" "SoHL" "thalorna" "Thalorna" "kethira" "Kethira" "hm3" "HM3" -}}
  1. Home
    2. {{- if and (ne $type "") (not $isAggregator) -}} {{- $section := "" -}} {{- if eq $type "doc" -}} {{- $pkg := .Params.package | default "" -}} {{- $cat := .Params.category | default "" -}} {{- if and (ne $pkg "") (ne $cat "") -}} {{- $pkgLabel := index $packageLabels $pkg | default (title $pkg) -}} {{- $catLabel := replace $cat "-" " " | title -}} {{- $section = dict "label" (printf "%s %s" $pkgLabel $catLabel) "url" (printf "/%s/%s/" $pkg $cat) -}} {{- end -}} {{- else -}} {{- $section = index $compoundLabels $type -}} {{- end -}} {{- if $section -}}
  2. {{ $section.label }}
    3. {{- else -}} {{- $typePage := site.GetPage (printf "/type/%s" $type) -}} {{- if $typePage -}}
  3. {{ $typePage.Title }}
    4. {{- else -}}
  4. {{ $type }}
    5. {{- end -}} {{- end -}} {{- end -}}
  5. {{ .Title }}
